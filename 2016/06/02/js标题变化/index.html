Js标题变化

这里是写标题的动态状态的方法,比如有经常看到滚动或者闪烁的”您有新的消息”.


滚动的方式 :


setInterval("run()",100); //这里走一个周期 100毫秒执行一次run()

function run(){
        var title = document.title; //获取当前title的内容
        var firstString = title.charAt(0); //获取title内第一个字符
        var lastString = title.substr(1); //取title指定位置的内容(这里从第二位起截取)
        document.title = lastString+firstString; //赋新值将,刚刚截取的重新组合
    }

滑动的方式 :


 show(); //执行show;

function show(){
    document.title = "【新消息】中奖500万，记得纳税哦！"; //将title设置新定义一下.
    setTimeout("hidden()",500); //设定一个时间500毫秒后执行hidden();
}
function hidden(){
    document.title = "【　　　】中奖500万，记得纳税哦！";//将title设置新定义一下.
    setTimeout("show()",500);//设定一个时间500毫秒后执行show();

}
